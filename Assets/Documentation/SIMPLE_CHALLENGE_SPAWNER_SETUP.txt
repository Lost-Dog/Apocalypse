using UnityEngine;

public class MyMissionScript : MonoBehaviour
{
    void SpawnMissionEnemies()
    {
        // Define what you want to spawn
        var requirements = new SimpleChallengeSpawnerManager.ChallengeRequirements
        {
            patrolCount = 3,
            eliteCount = 2,
            bossCount = 1,
            zombieCount = 0
        };

        // Spawn at random zone
        var result = SimpleChallengeSpawnerManager.Instance.SpawnChallenge(requirements);

        if (result != null)
        {
            Debug.Log($"Spawned at zone: {result.zone.zoneName}");
            Debug.Log($"Total enemies: {result.spawnedEnemies.Count}");
            
            // Store result to clear later
            currentMissionSpawn = result;
        }
    }

    void ClearMission()
    {
        SimpleChallengeSpawnerManager.Instance.ClearSpawnedEnemies(currentMissionSpawn);
    }
}
