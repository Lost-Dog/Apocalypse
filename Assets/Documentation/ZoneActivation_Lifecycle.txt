1. Get All Mission Zones

   â”‚   â””â”€ distance = Vector3.Distance(P, zone.position)
   â””â”€ Skip zones where distance > zone.zoneRadius

â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€

MissionZone FindZone(Vector3 pos, ChallengeType type)
{
    zones = FindAllZones(includeInactive: true);
    
    nearestZone = null;
    nearestDistance = infinity;
    
    foreach (zone in zones)
    {
        if (zone.isActive) continue;        // Skip active
        if (zone.type != type) continue;    // Skip wrong type
        
        distance = Distance(pos, zone.position);
        
        if (distance < zone.radius && distance < nearestDistance)
        {
            nearestZone = zone;
            nearestDistance = distance;
        }
    }
    
    return nearestZone;
}
